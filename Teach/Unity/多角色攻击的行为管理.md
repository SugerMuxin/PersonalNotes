

>当游戏中的英雄角色足够多的时候，可能很多的角色的技能和特效都是同质化的，那么这些同质化的技能的具体实现其实是可以共用的，这种情况下就可以将这些技能的具体实现抽出来单独的实现。后面哪个角色需要哪些技能就可以将这些技能的实现作为组件的形式去配置，甚至可以自由配置在角色的表中使用反射的方式来实现。


```
public abstract class Monster{

	public virtual void Attack(){
		BehaviourManager.Instance.Attack1(this.gameObject);
	}
	
}


public class MonsterA:Monster{
	
	public override void Attack(){
		BehaviourManager.Instance.Attack2(this.gameObject);
	}
	
}


public class MonsterB{

	public override void Attack(){
		
	}
	
}


public class BehaviourManager(){

	public static BehaviourManager Instance;

	public void Attack1(GameObject obj){
		
	}

	public void Attack2(GameObject obj){
		
	}

}

```
